Frame Template

@anabel.template(3)
def beam2d_template(
    q0: array = None,
    E:  float = None,
    A:  float = None,
    I:  float = None,
    L:  float = None
):
    def beam(v, q, state=None, E=E,A=A,I=I,L=L):
        C0 = E*I/L
        C1 = 4.0*C0
        C2 = 2.0*C0
        k = anp.array([[E*A/L,0.0,0.0],[0.0,C1,C2],[0.0,C2,C1]])
        return v, k@v, state
    return locals()

# create a model parameter from this expression
A  = model.expr(area, tf, bf)
